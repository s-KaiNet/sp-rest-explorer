<objective>
Fix path search to use substring matching instead of MiniSearch tokenization.

Purpose: UAT revealed two major issues — (1) slash queries like "web/lists" should match the literal contiguous substring, not independent tokens; (2) space queries like "lists get" return no results because MiniSearch's `/`-only tokenizer treats "lists get" as a single unrecognizable token. Both stem from MiniSearch being the wrong tool for path search. Replace it with direct substring filtering on the stored PathSearchDocument array.

Output: Working path search for both slash and space triggers, with correct highlighting.
</objective>

  3. **Create a `searchPathDocuments` function** that does the actual filtering:
     ```ts
     export function searchPathDocuments(query: string, limit = 50): PathSearchDocument[] {
       if (pathDocuments.length === 0) return []

       const mode = detectSearchMode(query)

       if (mode === 'name') return [] // shouldn't happen, but guard

       if (query.includes('/')) {
         // SLASH MODE: contiguous substring match
         // Strip leading _api/ from query if present (user might type "_api/web/lists")
         const normalizedQuery = query.replace(/^_api\//i, '').toLowerCase()
         if (normalizedQuery.length === 0) return []

         return pathDocuments
           .filter(doc => {
             const normalizedPath = doc.path.replace(/^_api\//i, '').toLowerCase()
             return normalizedPath.includes(normalizedQuery)
           })
           .slice(0, limit)
       }

       // SPACE MODE: every space-separated term must appear as substring (AND semantics)
       const terms = query.toLowerCase().split(/\s+/).filter(Boolean)
       if (terms.length === 0) return []

       return pathDocuments
         .filter(doc => {
           const lowerPath = doc.path.toLowerCase()
           return terms.every(term => lowerPath.includes(term))
         })
         .slice(0, limit)
     }
     ```

     Key design points:
     - `limit` defaults to 50 to prevent flooding the UI with thousands of matches (62K docs)
     - Slash mode: literal contiguous substring on the path (after stripping `_api/` prefix from both query and path for ergonomics)
     - Space mode: AND over all space-separated terms — each must appear as a substring somewhere in the path
     - Results returned in natural order (document insertion order = tree-walk order)

     The current `HighlightedPath` splits query into tokens on `/` and space, then checks if each path segment starts-with or includes any token. This is per-segment highlighting which doesn't match the contiguous substring semantics of slash mode.

     Replace the implementation with mode-aware highlighting:

     ```tsx
     function HighlightedPath({ path, query }: { path: string; query: string }) {
       if (!query || query.length < 3) return <span>{path}</span>

       if (query.includes('/')) {
         // SLASH MODE — highlight the contiguous substring match
         const normalizedQuery = query.replace(/^_api\//i, '').toLowerCase()
         const lowerPath = path.toLowerCase()
         // Find match position (strip _api/ from path for matching, but render full path)
         const pathWithoutPrefix = path.replace(/^_api\//i, '')
         const lowerPathWithoutPrefix = pathWithoutPrefix.toLowerCase()
         const matchIdx = lowerPathWithoutPrefix.indexOf(normalizedQuery)

         if (matchIdx === -1) return <span>{path}</span>

         // Calculate offset in full path (account for _api/ prefix if present)
         const prefixLen = path.length - pathWithoutPrefix.length
         const absStart = prefixLen + matchIdx
         const absEnd = absStart + normalizedQuery.length

         return (
           <span>
             {path.slice(0, absStart)}
             <strong className="font-bold text-foreground">{path.slice(absStart, absEnd)}</strong>
             {path.slice(absEnd)}
           </span>
         )
       }

       // SPACE MODE — highlight each matching term independently
       const terms = query.toLowerCase().split(/\s+/).filter(Boolean)
       const lowerPath = path.toLowerCase()

       // Build a highlight map (which char positions to bold)
       const highlights = new Array(path.length).fill(false)
       for (const term of terms) {
         let startPos = 0
         while (startPos < lowerPath.length) {
           const idx = lowerPath.indexOf(term, startPos)
           if (idx === -1) break
           for (let i = idx; i < idx + term.length; i++) highlights[i] = true
           startPos = idx + 1
         }
       }

       // Render with highlight spans
       const parts: Array<{ text: string; highlight: boolean }> = []
       let i = 0
       while (i < path.length) {
         const isHl = highlights[i]
         let j = i
         while (j < path.length && highlights[j] === isHl) j++
         parts.push({ text: path.slice(i, j), highlight: isHl })
         i = j
       }

       return (
         <span>
           {parts.map((part, idx) =>
             part.highlight ? (
               <strong key={idx} className="font-bold text-foreground">{part.text}</strong>
             ) : (
               <span key={idx}>{part.text}</span>
             )
           )}
         </span>
       )
     }
     ```

     This ensures:
     - Slash queries: the exact contiguous match is highlighted as one span (e.g., "web/Lists" is bolded as a unit)
     - Space queries: each term is highlighted wherever it appears in the path
